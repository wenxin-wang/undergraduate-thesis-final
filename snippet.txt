#!/bin/sh
iptables -t mangle -A PREROUTING -i eth2 -s 10.129.128.8 -j MARK --set-mark 1
iptables -t mangle -A PREROUTING -i eth2 -s 10.129.128.9 -j MARK --set-mark 2

ip route add 10.129.128.0/17 dev eth2 table 2
